```
dwm
├─ теги
│  ├─ cmd + number - перейти до вибраного тегу
│  │  └─ click on number
│  ├─ cmd + shift + number - перекинути активне вікно в вибраний тег
│  │  └─ cmd + click on number
│  ├─ cmd + control + number - обєднує вікна вибраних тегів
│  ├─ cmd + control + shift + number - відобразити активне вікно в вибраних тегах
│  ├─ cmd + 0 - переглянути всі вікна
│  ├─ cmd + shift + 0 - відобразити вікно в всіх тегах
│  ├─ cmd + k - перейти на наступний тег
│  └─ cmd + j - перейти на задній тег
├─ вікна
│  ├─ cmd + shift + k - перейти на наступне вікно
│  ├─ cmd + shift + j - перейти на заднє вікно
│  ├─ cmd + d - закрити вікно
│  ├─ cmd + n - режим класичний відображення вікон []=
│  │  ├─ cmd + l - збільшити ширину головної зони
│  │  ├─ cmd + h - зменшити ширину головної зони
│  │  ├─ cmd + shift + l - перейти на наступну розкладку вікон
│  │  ├─ cmd + shift + h - перейти на задню розкладку вікон
│  │  └─ cmd + return - перемістити вікно в головну зону
│  ├─ cmd + b - режим вільний відображиння вікон ><>
│  │  ├─ cmd + click on window - перемістити вікно
│  │  └─ cmd + shift + click on window - змінити розмір вікна
│  ├─ cmd + m - режим монокля відображиння вікон [M]
│  └─ cmd + shift + space - вести або вивести вікно з схеми вікон
├─ реголювання
│  ├─ cmd + space - змінювати розкладку клавіатури між українською і англійською
│  ├─ звук
│  │  ├─ cmd + + - збільшити звук
│  │  └─ cmd + - - зменшити звук
│  ├─ яскравість екрану
│  │  ├─ cmd + shift + + - збільшити яскравість
│  │  └─ cmd + shift + - - зменшити яскравість
│  ├─ яскравість клавіатури
│  │  ├─ cmd + control + + - збільшити яскравість
│  │  └─ cmd + control + - - зменшити яскравість
│  ├─ музика
│  │  ├─ option/alt + 1-6 - включити
│  │  └─ option/alt + 0 - виключити
│  ├─ звук клавіш
│  │  ├─ cmd + o - включити
│  │  └─ cmd + shift + o - виключити
│  └─ блютуз
│     ├─ cmd + shift + b - включити
│     └─ cmd + control + b - виключити
├─ програми
│  ├─ cmd + shift + return - st
│  ├─ cmd + shift + w - qutebrowser
│  ├─ cmd + shift + m - btop
│  ├─ cmd + shift + p - pacmixer
│  ├─ cmd + shift + g - steam
│  └─ control + space - відкрити меню програм
├─ cmd + g - сховати або показати статус бар
├─ cmd + q - вийти з графічного інтерфейсу X
└─ control + c,v - скопіювати, вставити
```
